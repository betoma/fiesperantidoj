
\subsection{\phipa{l} becomes \bripa{w} when it follows a vowel and does not precede a vowel or glide}

\begin{center}
l $\to$ w / V\_C , V\_\#
\end{center}

e.g., \espq{bulbo} \bripa{\prstr bul.b\schwa} $\to$ \bripa{\prstr buw.b\schwa}

\subsection{When \phipa{r} follows a vowel and does not preced a vowel, it rhotacizes the preceding vowel and disappears}

\begin{center}
    Vr $\to$ V\rhot{} / \_C , \_\#
\end{center}

e.g., \espq{forkuri} \bripa{f\schwa\alvrap\prstr ku.\alvrap i} $\to$ \bripa{f\schwa\rhot\,\prstr ku.\alvrap i}

\subsection{{\sc Lowlands:} After a nasal schwa, voiceless stops and affricates become voiced stops and affricates, voiced stops and affricates become nasal stops}

\begin{center}
    b d \paljstop{} g d\tiebar z d\tiebar\ezh{} \glotstop{} $\to$ m n \egna{} \engma{} n \egna{} / \~{\schwa}\_
\end{center}

e.g. \espq{embrio} \bripa{\~{\schwa}\prstr b\alvrap y.\paljfric\schwa} $\to$ \bripa{\~{\schwa}\prstr m\alvrap y.\paljfric\schwa}

\subsection{After a rhotic schwa, coronals become retroflex}

\begin{center}
    t d s z $\to$ \rette{} \retde{} \retesh{} \retezh{} / \schwa\rhot\_ \\[0.1cm]
    t\tiebar s t\tiebar\esh{} $\to$ t\tiebar\retesh / \schwa\rhot\_ \\[0.1cm]
    d\tiebar z t\tiebar\ezh{} $\to$ d\tiebar\retezh / \schwa\rhot\_ 
\end{center}

e.g., \espq{lerteco} \bripa{l\schwa\rhot\,tøt\tiebar s\schwa} $\to$ \bripa{l\schwa\rhot\,\rette øt\tiebar s\schwa}

\subsection{Schwas are deleted whenever possible (unless it would result in a truly untenable cluster)}

e.g., \espq{Esperanto} \bripa{\scstr e.sp\schwa\prstr r\~{\ahoh}.t\schwa} $\to$ \bripa{e\prstr spr\~{\ahoh}t}

\subsection{Liquids become glides after sonorants}

\begin{center}
    l r $\to$ j w / S\_
\end{center}

e.g., \espq{malebla} \bripa{\prstr mlebl} $\to$ \bripa{mjebl}

\subsection{Front vowels are raised and other vowels fronted after palatal consonants}

\begin{center}
    \begin{tabular}{@{}rrrl@{}}
        ø u $\to$ y & o $\to$ ø & \ahoh $\to$ \oesh &\multirow{2}{*}{/ \{c, \paljstop, \paljfric, j\}\_} \\
         a $\to$ \aesh & \aesh{} \schwa{} $\to$ e & e $\to$ i & 
    \end{tabular}
\end{center}

e.g., \espq{biologio} \bripa{bjol\prstr\paljstop y\paljfric} $\to$ \bripa{bjøl\prstr\paljstop y\paljfric}

\subsection{Vowels become nasalized after glottals}

\begin{center}
    V $\to$ \~{V} / \{\glotstop,h,\voih\}\_
\end{center}

A non-phonemic tonal downstep is also applied to the relevant syllable. Ordinarily, this non-phonemic downstep happens alongside the syllable marked with primary stress.

e.g., \espq{kohorto} \bripa{kho\rhot\,t} $\to$ \bripa{kh\~{o}\rhot\,t}, \espq{viandhaketa\^{j}o} \bripa{vj\~{\aesh}d.hak\prstr t\ahoh\ezh} $\to$ \bripa{vj\~{\aesh}d\downstep h\~{a}k\prstr t\ahoh\ezh}
